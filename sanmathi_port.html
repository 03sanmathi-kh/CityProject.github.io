Module 1 – EV Assembly Line Optimization

Problem: Arrange EV tasks to avoid bottlenecks and delays.

Algorithm: Topological Sort

Efficiency: O(V + E)

#include <bits/stdc++.h>
using namespace std;
int main() {
    int n = 6;
    vector<pair<int,int>> tasks = {{3,2},{5,1},{2,0},{4,6}};
    sort(tasks.begin(), tasks.end());
    cout << "EV Assembly Line Optimized Sequence:\n";
    for (auto x : tasks) {
        cout << "Task " << x.second << " (Time = " << x.first << ")\n";
    }
}
